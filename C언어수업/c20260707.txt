#include <stdio.h> // 누가 만들어놓은 소스코드 파일
/* main 함수 := 기계
* 역할: 프로그램 시작할때 무조건 실행해
* JAVA,C,C++,C#   main 함수 있어야함
* python, JS, TS main 필요 없음
* 
* 함수는 매개변수와 리턴타입
* 
* 모든 프로그램은 main 함수부터 찾아서 실행한다
*/
void main() {
	//print(f"ㅎㅇ")
	printf("ㅎㅇ");
	double b = 1.1; // 실수
	char c = 'A'; // 문자
	char* str1="ㅎㅇ"; // 문자열, 포인터, 배열

	/* C,JAVA
	한 scope에 같은 이름의 변수가 또 선언되면 안된다 */
	int a = 1; // 정수 4byte

	/* 이미 만들어놓은 공간 a 를 재사용하고 3을 저장할거야 */
	a = 3;

	/* Escape Squence:= 특수한 기호를 표시할때 사용
	엔터키 위에있는 \ 같은걸 사용함 
	그냥 문자로 취급하지 하고 특수한 문자로 강제로 만들어줌
	\n := 다음줄 개행 */
	// "a" \ "b"
	printf(" \n \"a\" \\ \"b\" \n");
	printf("a:%d \n",a);

}
