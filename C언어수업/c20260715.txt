#include <stdio.h>

/* 자료 결합도
원본 데이터에 손상이 안감 
c언어 에서는 숫자, 문자, 구조체 얘네는 자료 결합도 
나머지는 stamp 결합도 */
void swap(int a, int b) {
    int t = a;
    a = b;
    b = t;
    a = 99999;
}

int main() {
    int a = 1;
    int b = 2;
    swap(a, b);
    printf("a: %d, b: %d \n", a, b);

    return 0;
}

#include <stdio.h>
/* stamp 결합도
2번째로 좋은 결합도. 컴퓨터가 가장 좋아하는 방식 
사람이 제일 싫어하는 방식 
stamp 결합도는 원본을 조작함 */
void swap(int* a, int* b) {
    int t = *a;
    *a = *b;
    *b = t;
}

int main() {
    int a = 1;
    int b = 2;
    swap(&a, &b);
    printf("a: %d, b: %d \n", a, b);

    return 0;
}

#include <stdio.h>

int main() {
    int arr[3][2] = {
        {1,2},
        {3,4},
        {5,6}
    };
    arr[1][0]; // 3
    printf("%d \n", *(*(arr + 1) + 0));
    for (int a = 0; a < 3;a++) {
        for (int b = 0; b < 2;b++) {
            //printf("arr[%d][%d]=%d \n", a, b, arr[a][b]);
            printf("*(*(arr + %d) + %d)=%d \n", a, b, *(*(arr + a) + b));
        }
    }
    return 0;
}
