#include <stdio.h>

void main() {
    int a = 1;
    int b = 3;

    /* a = a + 2 
    a= */ 
    a += 2;

    /* a= a + 2 + b
       a= 1 + 2 + 3 */
    a += 2 + b;

    /* a= a*2 */
    a *= 2;


    a = 1; b = 3;

    /* 전위 연산자 
    뭐가 있건간에, a가 있는 장소 찾아가서 1 증가 영구 반영해 */ 
    ++a;

    /* 후위 연산자 
    뭐가 있건간에 a 는 다른거 먼저 다 하고 나중에 증가시켜 */
    a++;
    printf("%d,%d \n", a, b);

    a = 1; b = 3;

    /* a가 있는곳 찾아가서 값 증가시키기 영구반영 2번하고,
    그다음에 a= a+a 진행 한다 */
    a = ++a + ++a;
    printf("a:%d \n", a);

    a = 1; b = 3;
    int c = a++ + b;
    printf("c:%d \n", c);

    a = 1; b = 3;
    printf("a:%d,b:%d \n", a++, ++b);
    printf("a:%d,b:%d \n", a, b);

    return;
}

#include <stdio.h>

void main() {
    int rank = 10;
    int reward = 0;

    //       10
    switch (rank) {
        case 1:
            reward += 100;
            break;
        /* 똑같이 생긴거 선택. */
        case 2:
            reward += 150;
            // break 만나면 switch 빠져나옴
            break;
        default:
            reward = 50;
    }
    printf("rank:%d, reward:%d \n", rank, reward);
    return;
}

#include <stdio.h>

void main() {
    /* main scope 에 있는 a */
    int a = 1;
    if (a) {
        /* 새끼 if문 안에 있는 a
        * main 의 a랑 아무 관련 없음
        */
        int a = 5;
        printf("a: %d \n", a);
    }

    /* 눈때중으론 제일 가까운 a가 if문 안에있는 a 인데
    프로그램 세계에선 부모는 자식꺼 함부로 못갔다써요.
    그래서 그냥 main scope에 있는 a 갔다 보여준 거에요 */
    printf("a: %d \n", a);
    return;
}
