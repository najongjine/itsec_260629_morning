#include <stdio.h>

void main() {
    int a = 1;
    int b = 3;

    /* a = a + 2 
    a= */ 
    a += 2;

    /* a= a + 2 + b
       a= 1 + 2 + 3 */
    a += 2 + b;

    /* a= a*2 */
    a *= 2;


    a = 1; b = 3;

    /* 전위 연산자 
    뭐가 있건간에, a가 있는 장소 찾아가서 1 증가 영구 반영해 */ 
    ++a;

    /* 후위 연산자 
    뭐가 있건간에 a 는 다른거 먼저 다 하고 나중에 증가시켜 */
    a++;
    printf("%d,%d \n", a, b);

    a = 1; b = 3;

    /* a가 있는곳 찾아가서 값 증가시키기 영구반영 2번하고,
    그다음에 a= a+a 진행 한다 */
    a = ++a + ++a;
    printf("a:%d \n", a);

    a = 1; b = 3;
    int c = a++ + b;
    printf("c:%d \n", c);

    a = 1; b = 3;
    printf("a:%d,b:%d \n", a++, ++b);
    printf("a:%d,b:%d \n", a, b);

    return;
}
