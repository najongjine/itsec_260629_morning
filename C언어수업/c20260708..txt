#include <stdio.h>

void main() {
    int a = 11;
    // %d := 정수
    printf("a: %d \n", a);
   
    // %x := 16진수
    printf("a: %x \n", a);
    
    double d1 = 1.1;
    // %lf := 소수점
    printf("d1: %lf \n", d1);

    a = 65;
    // %c := 문자
    printf("a: %c \n", a);

    char* str = "abc";
    // %s := 문자열(문자가 여러개 모인것)
    printf("str: %s \n", str);


    // %o := 8진수
    printf("a: %o \n", a);

    // (재미용)메모리 주소는 16진수로 표현되요
    printf("a: %p \n", &a);

    d1 = 1.1;
    printf("소수를 정수로 강제출력: %d \n", d1);
    printf("str를 정수로 강제출력: %d \n", str);
}

#include <stdio.h>

void main() {
	int a = 1;

	++a; // a=a+1;
	a = 1 + 1 * 2;

	double d1 = 1 % 3; // 나누기 한다음 나머지
	printf("d1: %lf \n", d1);

	/*10진수 2를 2진수로 바꾸면 10 이 됩니다 */
	a = 2;
	a = a >> 1; // 2진수를 오른쪽으로 1칸 밀어라. 오른쪽엔 낭떠러지 있다

	a = 2;
	a = a << 1; // 2진수를 왼쪽으로 1칸 밀어라

	/* 비교 연산자
	아쉽게도 c언어에선 true, false 데이터 타입이 없어서
	0, null, \0  얘네는 false, 나머지는 다 true */
	a = 1 >= 3;

	/* 3항 연산자 
	1>3 물어보고, true면 왼쪽꺼, false면 오른쪽꺼 가져갑니다 */
	a = 1 > 3 ? 1 : 0;

	a = 2 & 1; // AND bit 연산
	a = 2 | 1; // OR bit 연산
	a = 2 ^ 1; // XOR 연산

	a = 1 && 1; // 논리 AND 연산
	a = 1 || 0; // 논리 OR 연산

}
