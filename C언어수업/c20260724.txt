// 선택정렬
#include <stdio.h>

int main() {
    int arr[] = { 64, 25, 12, 22, 11 };
    int size = 5;

    for (int i = 0; i < size - 1; i++) {
        int minIndex = i;

        // i번째 이후에서 가장 작은 값의 위치 찾기
        for (int j = i + 1; j < size; j++) {
            if (arr[j] < arr[minIndex]) {
                minIndex = j;
            }
        }

        // i번째 값과 가장 작은 값 교환
        int temp = arr[i];
        arr[i] = arr[minIndex];
        arr[minIndex] = temp;
    }

    for (int i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }

    return 0;
}

//삽입정렬
#include <stdio.h>

int main() {
    int arr[] = {64, 25, 12, 22, 11};
    int size = 5;

    for (int i = 1; i < size; i++) {
        int key = arr[i];     // 삽입할 값
        int j = i - 1;

        // key보다 큰 값들을 오른쪽으로 한 칸씩 이동
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }

        // 알맞은 위치에 key 삽입
        arr[j + 1] = key;
    }

    for (int i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }

    return 0;
}

//퀵정렬

#include <stdio.h>

void quickSort(int arr[], int left, int right) {
    int i = left;
    int j = right;
    int pivot = arr[left];
    int temp;

    while (i <= j) {
        // 피벗보다 작은 값은 건너뛴다.
        while (arr[i] < pivot) {
            i++;
        }

        // 피벗보다 큰 값은 건너뛴다.
        while (arr[j] > pivot) {
            j--;
        }

        // 잘못 배치된 두 값을 교환한다.
        if (i <= j) {
            temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;

            i++;
            j--;
        }
    }

    // 왼쪽 부분 정렬
    if (left < j) {
        quickSort(arr, left, j);
    }

    // 오른쪽 부분 정렬
    if (i < right) {
        quickSort(arr, i, right);
    }
}

int main() {
    int arr[] = { 5, 3, 8, 4};
    int size = sizeof(arr) / sizeof(arr[0]);

    quickSort(arr, 0, size - 1);

    for (int i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }

    return 0;
}
