#include <stdio.h>

void main() {
	int a = 1;
	int b = a;
	b = 3;

	a * b; // 곱하기
	a & b; // 비트 AND 연산
	/* data type 옆에 * 있으면 포인터
	데이터 딸랑 하나 왼쪽에 & 있으면 주소뽑기 */
	int* pa = &a;
	printf("a:%d, b:%d \n", a, b);
	printf("pa:%p \n", pa);

	/* 변수 왼쪽에 *가 붙으면 주소보고 원산지 찾아가기가 됩니다
	= 사인 기준으로, = 사인 왼쪽은 원산지 찾아가서 거기에 저장하기
	그외는 원산지 찾아가서 값 복붙해오기가 됩니다 */
	/* 포인터가 어려운 이유는 * 기호가 언제는 포인터 변수선언
	언제는 곱하기
	언제는 원산지 찾아가서 가져오거나 저장하기 때문에
	헷갈립니다 */
	// *pa = 1 * 2
	*pa = *pa * 2;
	printf("a:%d, b:%d \n", a, b);


}
